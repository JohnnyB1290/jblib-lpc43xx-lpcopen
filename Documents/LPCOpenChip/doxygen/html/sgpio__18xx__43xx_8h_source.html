374 typedef uint32_t SGPIO_SliceMuxConfig;
375 /* function to make the slice mux register contents */
376 STATIC INLINE SGPIO_SliceMuxConfig SGPIO_makeSliceMuxConfig (
377  SMC_MatchMode matchMode,
378  SMC_ClkCaptureMode clkCaptureMode,
379  SMC_ClkGenMode clkGenMode,
380  SMC_InvOutClk invOutClk,
381  SMC_DataCapMode dataCaptureMode,
382  SMC_ParallelMode parallelMode,
383  SMC_InvQualifier invQualifier)
384 {
385  return (SGPIO_SliceMuxConfig)(
386  (matchMode << 0) |
387  (clkCaptureMode << 1) |
388  (clkGenMode << 2) |
389  (invOutClk << 3) |
390  (dataCaptureMode << 4) |
391  (parallelMode << 6) |
392  (invQualifier << 8)
393  );
394 }
