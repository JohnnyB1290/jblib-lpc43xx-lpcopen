474 {
475  /* Save MAC address */
476  pENET->MAC_ADDR0_LOW = ((uint32_t) macAddr[3] << 24) |
477  ((uint32_t) macAddr[2] << 16) | ((uint32_t) macAddr[1] << 8) |
